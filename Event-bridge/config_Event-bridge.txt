# **FULL AUTOMATED BACKUP SCRIPT FOR ENTIRE EVENTBRIDGE SETUP**  
This script gathers **EVERYTHING** in a single run.

### **backup-eventbridge.sh**
```bash
#!/bin/bash

DIR="eventbridge-backup-$(date +%F)"
mkdir -p $DIR

echo "=== Backing up AWS EventBridge Configuration to $DIR ==="

#################################
# EVENT BUSES
#################################
aws events list-event-buses > $DIR/event-buses.json

for bus in $(jq -r '.EventBuses[].Name' $DIR/event-buses.json); do
    aws events describe-event-bus --name "$bus" > $DIR/event-bus-$bus.json
done

#################################
# RULES AND TARGETS
#################################
aws events list-rules > $DIR/rules.json

for rule in $(jq -r '.Rules[].Name' $DIR/rules.json); do
    aws events describe-rule --name "$rule" > $DIR/rule-$rule.json
    aws events list-targets-by-rule --rule "$rule" > $DIR/rule-$rule-targets.json
done

#################################
# ARCHIVES
#################################
aws events list-archives > $DIR/archives.json

#################################
# API DESTINATIONS
#################################
aws events list-api-destinations > $DIR/api-destinations.json

#################################
# CONNECTIONS
#################################
aws events list-connections > $DIR/connections.json

#################################
# ENDPOINTS
#################################
aws events list-endpoints > $DIR/endpoints.json

#################################
# PARTNER SOURCES
#################################
aws events list-partner-event-sources > $DIR/partner-sources.json
aws events list-partner-event-source-accounts > $DIR/partner-accounts.json

#################################
# REPLAYS
#################################
aws events list-replays > $DIR/replays.json

echo "=== EventBridge Backup Completed Successfully ==="
